## Load packages
library(plot3D)

Square Max Matrix

max_matrix = function(n) {
  n <- 10
  max_m <- c()
  for(i in c(1:n)) {
    for (j in c(1:n)) {
      max_m <- append(max_m, max(i, j))
    }
  }
  dim(max_m) <- c(n, n)
  max_m
}
